142  // Checks for any events on the socket.
143  bool socket::poll()
144  {
145  bool result = false;
146  if (
147  (m_state != SOCKET_STATE_STATELESS)
148  and (m_state != SOCKET_STATE_LISTENING)
149  and (m_state != SOCKET_STATE_CONNECTED)
150  )
151  {
152  log::get()->error("Socket is not in a state in"
153  " which it can be polled.");
154  return result;
155  }
156  /* Call epoll_wait to fetch a bunch of events and
157  * loop through them to handle them. */
158  int event_count = epoll_wait(
159  m_epoll_fd, m_epoll_events, m_epoll_maxevents, -1);
160  for (int n_event = 0 ; n_event < event_count ; ++n_event)
161  {
162  int event_fd = m_epoll_events[n_event].data.fd;
163  if (
164  (m_epoll_events[n_event].events & EPOLLERR)
165  or (m_epoll_events[n_event].events & EPOLLHUP)
166  or (not (m_epoll_events[n_event].events & EPOLLIN))
167  )
168  {
169  if (event_fd == m_fd)
170  {
171  // Error on our socket. :(
172  log::get()->error("epoll error on the listening socket.");
173  close();
174  }
175  else if (m_state == SOCKET_STATE_LISTENING)
176  {
177  // I'm a server, and I received an error on a client
178  // socket. Consider connection dropped, so clean it up,
179  // and fire a disconnection event.
180  m_remove_peer(event_fd);
181  if (m_server != nullptr)
182  {
183  auto client_peer_it = m_peers.find(event_fd);
184  if (client_peer_it == m_peers.end())
185  {
186  std::string errstr = "Received an error on a ";
187  errstr += "file-descriptor for which there is no ";
188  errstr += "corresponding client. (fd=";
189  errstr += std::to_string(event_fd) + ")";
190  log::get()->error(errstr);
191  }
192  else
193  {
194  m_server->on_disconnect(client_peer_it->second);
195  m_remove_peer(client_peer_it->first);;
196  }
197  }
198  }
199  continue;
200  }
201  else if (m_epoll_events[n_event].data.fd == m_fd)
202  {
203  /* An event came in on our socket.
204  * This is typically a connection attempt.
205  * If we're in a listening state, handle incomming
206  * connections and fire an on_connect event. */
207  if (m_state == SOCKET_STATE_LISTENING)
208  {
209  peer* my_peer = m_try_accept();
210  if (my_peer != nullptr)
211  {
212  m_peers.insert({my_peer->get_fd(), my_peer});
213  if (m_server != nullptr)
214  {
215  m_server->on_connect(my_peer);
216  result = true;
217  }
218  }
219  }
220  }
221  else
222  {
223  /* An event came in on one of the peer sockets.
224  * This is typically peer data. */
225  auto client_peer_it = m_peers.find(
226  m_epoll_events[n_event].data.fd);
227  if (client_peer_it == m_peers.end())
228  {
229  std::string errstr = "Received an event with a file";
230  errstr += "descriptor number not found in the server's list";
231  errstr += " of peers. (fd = ";
232  errstr += std::to_string(m_epoll_events[n_event].data.fd);
233  errstr += ")";
234  return false;
235  }
236  if (m_state == SOCKET_STATE_LISTENING)
237  {
238  if (m_server != nullptr)
239  {
240  m_server->on_receive(client_peer_it->second);
241  }
242  }
243  result = true;
244  }
245  }
246  return result;
